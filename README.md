function drawRadar(){
  const canvas=document.getElementById('radarChart'); if(!canvas) return;
  const {ctx,w,h}=prepCanvas(canvas); ctx.clearRect(0,0,w,h);
  const rows=DATA.aggregate.rows; const n=rows.length; const cx=w/2, cy=h/2+8, R=Math.min(w,h)*0.35;

  ctx.strokeStyle='rgba(255,255,255,.12)'; ctx.lineWidth=1;
  for(let ring=1;ring<=5;ring++){
    ctx.beginPath();

function rowsForMode(mode){
  if(mode==='aggregate') return DATA.aggregate.rows.map(r=>({mode:'aggregate', categoryId:r.categoryId, category:r.category, short:r.short, weight:r.weight, aRaw:r.aRawAvg, bRaw:r.bRawAvg, aWeighted:r.aWeightedAvg, bWeighted:r.bWeightedAvg, margin:r.weightedMargin, winner:r.winner, rationale:'Aggregate average across all three model scorecards.'}));

function sortRows(rows){
  const s=document.getElementById('sortMode')?.value||'category';
  rows=[...rows];
  if(s==='marginDesc') rows.sort((a,b)=>b.margin-a.margin);
  else if(s==='closest') rows.sort((a,b)=>Math.abs(a.margin)-Math.abs(b.margin));
  else if(s==='weight') rows.sort((a,b)=>b.weight-a.weight || a.categoryId-b.categoryId);
  else rows.sort((a,b)=>a.categoryId-b.categoryId);
  return rows;
}

function startFxCanvas(){
  const c=document.getElementById('fxCanvas'); if(!c) return;
  const ctx=c.getContext('2d'); const reduce=window.matchMedia && matchMedia('(prefers-reduced-motion: reduce)').matches;
  let w=0,h=0,dpr=1,p=[];
  function size(){
    dpr=Math.min(devicePixelRatio||1,2); w=innerWidth; h=innerHeight;
    c.width=Math.floor(w*dpr); c.height=Math.floor(h*dpr); c.style.width=w+'px'; c.style.height=h+'px'; ctx.setTransform(dpr,0,0,dpr,0,0);
    const n=Math.max(24,Math.min(78,Math.floor(w*h/19000)));
    p=Array.from({length:n},(_,i)=>({x:Math.random()*w,y:Math.random()*h,r:Math.random()*1.7+.5,vx:(Math.random()-.5)*(w<700?.18:.3),vy:(Math.random()-.5)*(w<700?.18:.3),h:i%3}));
    draw(true);
  }
  function draw(once=false){
    ctx.clearRect(0,0,w,h);
    const g=ctx.createRadialGradient(w*.35,h*.15,0,w*.35,h*.15,Math.max(w,h));
    g.addColorStop(0,'rgba(63,244,255,.10)'); g.addColorStop(.5,'rgba(160,115,255,.05)'); g.addColorStop(1,'rgba(0,0,0,0)'); ctx.fillStyle=g; ctx.fillRect(0,0,w,h);
    const md=w<700?90:135;
    for(let i=0;i<p.length;i++){
      const a=p[i]; if(!once&&!reduce){a.x+=a.vx;a.y+=a.vy;if(a.x<0||a.x>w)a.vx*=-1;if(a.y<0||a.y>h)a.vy*=-1}
      for(let j=i+1;j<p.length;j++){const b=p[j],dx=a.x-b.x,dy=a.y-b.y,d=Math.hypot(dx,dy);if(d<md){ctx.globalAlpha=(1-d/md)*(w<700?.13:.2);ctx.strokeStyle=a.h===0?'#3ff4ff':a.h===1?'#42f5b6':'#a073ff';ctx.beginPath();ctx.moveTo(a.x,a.y);ctx.lineTo(b.x,b.y);ctx.stroke()}}
      ctx.globalAlpha=w<700?.35:.48;ctx.fillStyle=a.h===0?'#3ff4ff':a.h===1?'#42f5b6':'#ff4ecd';ctx.beginPath();ctx.arc(a.x,a.y,a.r,0,Math.PI*2);ctx.fill()
    }
    ctx.globalAlpha=1; if(!once&&!reduce) requestAnimationFrame(()=>draw(false));
  }
  size(); addEventListener('resize',()=>{clearTimeout(window.__fxr);window.__fxr=setTimeout(size,160)}); if(!reduce) requestAnimationFrame(()=>draw(false));
}
